## 2. Setup in Your Project

To keep your API key safe and out of sight, it's best to store it in environment variables. This way, your key stays hidden and secure, 
and you don't have to worry about it showing up in your code. Plus, it's super easy to manage different keys for different environments like development, staging, and production.

  ```go Go
  package main

  import (
      "fmt"
      "net/http"
      "io/ioutil"
      "os"
  )

  func main() {
      apiKey := os.Getenv("MUFI_API_KEY")
      client := &http.Client{}
      req, _ := http.NewRequest("GET", "https://api.example.com/endpoint", nil)
      req.Header.Add("Authorization", "Bearer " + apiKey)

      resp, _ := client.Do(req)
      body, _ := ioutil.ReadAll(resp.Body)
      fmt.Println(string(body))
  }
  ```

  ```java Java
  import java.io.IOException;
  import java.net.HttpURLConnection;
  import java.net.URL;
  import java.util.Scanner;

  public class Main {
      public static void main(String[] args) throws IOException {
          String apiKey = System.getenv("MUFI_API_KEY");
          URL url = new URL("https://api.example.com/endpoint");
          HttpURLConnection conn = (HttpURLConnection) url.openConnection();
          conn.setRequestMethod("GET");
          conn.setRequestProperty("Authorization", "Bearer " + apiKey);

          Scanner scanner = new Scanner(conn.getInputStream());
          while (scanner.hasNext()) {
              System.out.println(scanner.nextLine());
          }
          scanner.close();
      }
  }
  ```
